\def\eesCommentaryAfterToe{
\vspace*{5cm}
Almo factori omnium,\\
pio fautori hominum\\
laudes canamus hilares,\\
preces fundamus humiles.\\[1ex]
Cuius summa potentia\\
eduxit cuncta entia,\\
cuius appensa digito\\
utraque moles sistitur.\\[1ex]
Illius sapientia\\
aeternis adstans sedibus\\
associari hominum\\
sanctis ardescit mentibus.\\[1ex]
Qui hanc coepere cupidi\\
ducemque nacti fuerant,\\
Deum laudabunt fervidi\\
in sempiterna saecula.\\[1ex]
Amen.
}
